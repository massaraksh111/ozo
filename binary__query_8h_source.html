118  static auto make_impl(text_type text, const params_type& params,
119  const oid_map_type& oid_map, const buffer_allocator_type& buffer_allocator) {
120 
121  auto result = std::make_shared<impl_type>(std::move(text), buffer_allocator);
122 
123  ozo::detail::ostreambuf osbuf(result->buffer);
124  ozo::ostream os(&osbuf);
125 
126  const auto range = hana::to<hana::tuple_tag>(
127  hana::make_range(hana::size_c<0>, hana::size_c<params_count>));
128 
129  hana::for_each(range, [&] (auto field) {
130  field_proxy<field> proxy(*result, os);
131  write_meta(oid_map, params[field], proxy);
132  });
133 
134  std::size_t offset = 0;
135  hana::for_each(range, [&] (auto field) {
136  const auto size = result->lengths[field];
137  if (size && size != null_state_size) {
138  result->values[field] = std::data(result->buffer) + offset;
139  offset += size;
140  } else {
141  result->values[field] = nullptr;
142  }
143  }
144  );
145  return result;
146  }
